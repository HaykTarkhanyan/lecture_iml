\begin{enumerate}[a)]
	\item 	Odds for class 1 vs. class 0: $$\text{odds} = P(y=1)/P(y=0)$$
	Odds Ratio (relative chance): $$\text{odds ratio} = \frac{\text{odds}_{x_j+1}}{\text{odds}}$$
	\begin{align*}
		\text{odds ratio} &= \frac{\text{odds}_{x_j+1}}{\text{odds}} = \frac{P(y=1|x=1)/P(y=0|x=1)}{P(y=1|x=0)/P(y=0|x=0)} \\
		&= \frac{P(y=1|x=1)\cdot P(y=0|x=0)}{P(y=1|x=0)\cdot P(y=0|x=1)} \\
		&= \frac{a \cdot d}{c \cdot b} = \frac{218\cdot 85}{21 \cdot 45} = 19.6
	\end{align*}
	\textbf{Interpretation:} The risk of having a Hepatitis A infection is $19.6$ higher for people who have eaten salsa compared to people who have not eaten salsa (which is the reference). \\
\end{enumerate}
